函数声明式 

            {
                function fn(m,n){
                    return m**n
                }
                console.log(fn(12,3))
            }

函数表达式(匿名函数) 

            {
                var ff = function(m,n){
                    return m*n
                }
                console.log(ff(10,12)) // 120
            }

new Function() 

            // new Function()  函数体放在最后，所有的参数都写成字符串形式
            {
                let sum = new Function("m","n", "return m + n")
                console.log(sum(22,21)) // 43
            }
