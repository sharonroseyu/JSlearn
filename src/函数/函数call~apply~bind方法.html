call 改变 this 指向的方法

                var foo1 = {
                    name:"Lili",
                    logName:function(age){
                        console.log(this.name,age)  // LUCY 13
                    }
                }
    
                var boo1 = {
                    name: "LUCY",
                }
                // call 改变 this 指向
                foo1.logName.call(boo1,13)

apply 改变 this 指向的方法

                var foo2 = {
                    name:"张三",
                    logName:function(age){
                        console.log(this.name,age)  // 李四 18
                    }
                }
    
                var boo2 = {
                    name: "李四",
                }
                // aapply 改变 this 指向
                foo2.logName.apply(boo2,[18])

bind 改变 this 指向的方法

                var foo3 = {
                    name:"王五",
                    logName:function(age){
                        console.log(this.name,age)  // 马六 13
                    }
                }
    
                var boo3 = {
                    name: "马六",
                }
                // bind 改变 this 指向
                foo1.logName.bind(boo3,13)()
