高阶函数之 map 

                // map 高阶函数
                {
                    let arr = [1,2,3,4,5,6]
                    let res = arr.map((item)=>{
                        return item+10
                    })
                    console.log(res) //  [11, 12, 13, 14, 15, 16]
                }

                {
                    var kvarr = [
                        {key:1,value:10},
                        {key:2,value:15},
                        {key:3,value:20},
                        {key:4,value:25}
                    ]
                    var reskvarr = kvarr.map((item)=>{
                        var obj = {}
                        obj[item.key] = item.value
                        return obj;
                    })
                    console.log(reskvarr) // [{1: 10}, {2: 15}, {3: 20}, {4: 25}]
                }

高阶函数之 reduce 

                // map 高阶函数
                {
                    var arr = [10,20,30,40]
                    var sum = arr.reduce((pre,cur)=> pre + cur)
                    console.log(sum) // 100
                }
