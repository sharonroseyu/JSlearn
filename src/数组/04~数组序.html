数组去重方法一：利用indexOf()

            // 数组爱旭：
            {
                let arr = [3,5,6,7,8,1,4,3,1,5,1,7,8,9]
                console.log(`原始数组长度 ： `,arr.length)
                function uniqe(array){
                    let newarr = [];
                    for(let i = 0 ; i < array.length ; i++){
                        if(newarr.indexOf(array[i]) === -1){
                            newarr.push(array[i])
                        }
                    }
                    return newarr;
                }
                console.log(`去重后的数组 ： `,uniqe(arr))
                console.log(`去重后的数组的长度 ： `,uniqe(arr).length)
            }
