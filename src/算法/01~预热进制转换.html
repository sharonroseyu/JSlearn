            {
                // 进制转换
                // 将十进制转成八进制
                console.log(`十进制转成八进制 ： `,(100).toString(8)) // 144
                console.log(`十进制转成八进制 ： `,(9).toString(8))  // 11
                console.log(`十进制转成八进制 ： `,(16).toString(8)) //  20
        
                // 将十进制转成二进制
                console.log(`十进制转成二进制 ： `,(16).toString(2)) //  10000
                console.log(`十进制转成二进制 ： `,(3).toString(2)) //  11
        
                // 将十进制转成十六进制 1 2 3 4 5 6 7 8 9 a b c  d  e  f
                console.log(`十进制转成十六进制 ： `,(16).toString(16)) //  10
                console.log(`十进制转成十六进制 ： `,(17).toString(16)) //  11
                console.log(`十进制转成十六进制 ： `,(25).toString(16)) //  19
        
                // 将其他进制转成十进制
                console.group(`其他制转成十进制`)
                console.log(`二进制转成十进制 ： `, parseInt(10000,2)) //  10
                console.log(`八进制转成十进制 ： `, parseInt(`110`,8)) //  72
                console.log(`十六进制转成十进制 ： `, parseInt(`e`,16)) //  14
                console.groupEnd()
            }
