字面量方法声明的对象

                var obj = {name:"Lili",age:12}
                console.log(obj)
                console.log("obj.__proto__ : ",obj.__proto__)
                console.log("obj.__proto__.name : ",obj.__proto__.name) //  undefined
                console.log("obj.__proto__.constructor : ",obj.__proto__.constructor) // Object() { [native code]}

构造函数方法声明的对象

                var obj1 = new Object({name:"Lili",y:13});
                console.log(obj1); 
                console.log(obj1.name); //Lili
                console.log(obj1.__proto__.name); // undefined
                console.log(obj1.__proto__.name === obj1.name); // false

内置方法 Object.create 方法声明的对象

                let obj2 = Object.create({color:"red",seed: "133km/h"})
                console.log(obj2) // {}
                console.log(obj2.__proto__)  // {color: "red", seed: "133km/h"}
                console.log(obj2.color)  // red
                console.log(obj2.__proto__.color )  // red
                console.log(obj2.__proto__.color === obj2.color )  // true

以上三种方法的优缺点

                1.功能:都能实现对象的声明,并能够赋值和取值
                2.继承性:内置方法创建的对象继承到__proto__属性上
                3.隐藏属性:三种声明方法会默认为内部的每个成员（属性或方法）生成一些隐藏属性，这些隐藏属性是可以读取和可配置的,属性分类见下面
                4.属性读取:Object.getOwnPropertyDescriptor()或getOwnPropertyDescriptor()
                5.属性设置:Object.definePropertype或Object.defineProperties
