遍历多层对象

                function traverse(data){
                    // 如果不存在则直接返回
                    if(!data || !data.length){
                        return;
                    }
        
                    for(let i = 0 ; i < data.length ; i++){
                        var childs = data[i].children;
                        console.log("遍历 ： ",data[i].key," ======== " ,data[i]["name"]);
                        if(childs && childs.length > 0){
                            traverse(childs)
                        }
                    }
                }
                traverse(data)
